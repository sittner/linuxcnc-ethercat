109 static int export_std_pins(struct lcec_slave *slave, ec_pdo_entry_reg_t *pdo_entry_regs, int pid, hal_bit_t **pin, hal_pin_dir_t dir) {
110  lcec_master_t *master = slave->master;
111  lcec_slave_modparam_t *p;
112  int count, err;
113 
114  for (p = slave->modparams, count = 0; p != NULL && p->id >= 0; p++) {
115  // skip not matching params
116  if (p->id != pid) {
117  continue;
118  }
119 
120  // export pin
121  if ((err = lcec_pin_newf(HAL_BIT, dir, (void *) pin, "%s.%s.%s.%s", LCEC_MODULE_NAME, master->name, slave->name, p->value.str)) != 0) {
122  return err;
123  }
124 
125  // next item
126  pin++;
127  count++;
128  }
129 
130  return count;
131 }

133 static int lcec_el1918_logic_preinit(struct lcec_slave *slave) {
134  lcec_master_t *master = slave->master;
135  lcec_slave_modparam_t *p;
136  int index, stdin_count, stdout_count;
137  struct lcec_slave *fsoe_slave;
138  const LCEC_CONF_FSOE_T *fsoeConf;
139 
140  slave->pdo_entry_count = LCEC_EL1918_LOGIC_PDOS;
141 
142  stdin_count = 0;
143  stdout_count = 0;
144  for (p = slave->modparams; p != NULL && p->id >= 0; p++) {
145  switch(p->id) {
146  case LCEC_EL1918_LOGIC_PARAM_SLAVEID:
147  // find slave
148  index = p->value.u32;
149  fsoe_slave = lcec_slave_by_index(master, index);
150  if (fsoe_slave == NULL) {
151  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "%s.%s: slave index %d not found\n", master->name, slave->name, index);
152  return -EINVAL;
153  }
154 
155  fsoeConf = fsoe_slave->fsoeConf;
156  if (fsoeConf == NULL) {
157  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "%s.%s: slave index %d is not a fsoe slave\n", master->name, slave->name, index);
158  return -EINVAL;
159  }
160 
161  slave->pdo_entry_count += LCEC_EL1918_LOGIC_PARAM_SLAVE_PDOS + LCEC_EL1918_LOGIC_PARAM_SLAVE_CH_PDOS * fsoeConf->data_channels;
162  break;
163 
164  case LCEC_EL1918_LOGIC_PARAM_STDIN_NAME:
165  stdin_count++;
166  if (stdin_count > LCEC_EL1918_LOGIC_DIO_MAX_COUNT) {
167  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "%s.%s: maximum stdin count exceeded.\n", master->name, slave->name);
168  return -EINVAL;
169  }
170 
171  break;
172 
173  case LCEC_EL1918_LOGIC_PARAM_STDOUT_NAME:
174  stdout_count++;
175  if (stdout_count > LCEC_EL1918_LOGIC_DIO_MAX_COUNT) {
176  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "%s.%s: maximum stdout count exceeded.\n", master->name, slave->name);
177  return -EINVAL;
178  }
179 
180  break;
181  }
182  }
183 
184  if (stdin_count > 0) {
185  slave->pdo_entry_count += LCEC_EL1918_LOGIC_STDIN_PDOS;
186  }
187  if (stdout_count > 0) {
188  slave->pdo_entry_count += LCEC_EL1918_LOGIC_STDOUT_PDOS;
189  }
190 
191  return 0;
192 }

194 int lcec_el1918_logic_init(int comp_id, struct lcec_slave *slave, ec_pdo_entry_reg_t *pdo_entry_regs) {
195  lcec_master_t *master = slave->master;
196  lcec_el1918_logic_data_t *hal_data;
197  lcec_el1918_logic_fsoe_t *fsoe_data;
198  lcec_slave_modparam_t *p;
199  int fsoe_idx, index, err;
200  lcec_el1918_logic_fsoe_crc_t *crc;
201  struct lcec_slave *fsoe_slave;
202  const LCEC_CONF_FSOE_T *fsoeConf;
203 
204  // initialize callbacks
205  slave->proc_read = lcec_el1918_logic_read;
206  slave->proc_write = lcec_el1918_logic_write;
207 
208  // count fsoe slaves
209  for (fsoe_idx = 0, p = slave->modparams; p != NULL && p->id >= 0; p++) {
210  if (p->id == LCEC_EL1918_LOGIC_PARAM_SLAVEID) {
211  fsoe_idx++;
212  }
213  }
214 
215  // alloc hal memory
216  if ((hal_data = hal_malloc(sizeof(lcec_el1918_logic_data_t) + fsoe_idx * sizeof(lcec_el1918_logic_fsoe_t))) == NULL) {
217  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "hal_malloc() for slave %s.%s failed\n", master->name, slave->name);
218  return -EIO;
219  }
220  memset(hal_data, 0, sizeof(lcec_el1918_logic_data_t));
221  hal_data->fsoe_count = fsoe_idx;
222  slave->hal_data = hal_data;
223 
224  // initialize POD entries
225  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0xf100, 0x01, &hal_data->state_os, NULL);
226  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0xf100, 0x02, &hal_data->cycle_counter_os, NULL);
227 
228  // export pins
229  if ((err = lcec_pin_newf_list(hal_data, slave_pins, LCEC_MODULE_NAME, master->name, slave->name)) != 0) {
230  return err;
231  }
232 
233  // map and export stdios
234  hal_data->std_in_count = export_std_pins(slave, pdo_entry_regs, LCEC_EL1918_LOGIC_PARAM_STDIN_NAME, hal_data->std_in_pins, HAL_IN);
235  if (hal_data->std_in_count < 0) {
236  return hal_data->std_in_count;
237  }
238  if (hal_data->std_in_count > 0) {
239  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0xf788, 0x00, &hal_data->std_in_os, NULL);
240  }
241 
242  hal_data->std_out_count = export_std_pins(slave, pdo_entry_regs, LCEC_EL1918_LOGIC_PARAM_STDOUT_NAME, hal_data->std_out_pins, HAL_OUT);
243  if (hal_data->std_out_count < 0) {
244  return hal_data->std_out_count;
245  }
246  if (hal_data->std_out_count > 0) {
247  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0xf688, 0x00, &hal_data->std_out_os, NULL);
248  }
249 
250  // map and export fsoe slave data
251  for (fsoe_idx = 0, fsoe_data = hal_data->fsoe, p = slave->modparams; p != NULL && p->id >= 0; p++) {
252  if (p->id == LCEC_EL1918_LOGIC_PARAM_SLAVEID) {
253  // find slave
254  index = p->value.u32;
255  fsoe_slave = lcec_slave_by_index(master, index);
256  fsoe_data->fsoe_slave = fsoe_slave;
257  fsoe_slave->fsoe_slave_offset = &fsoe_data->fsoe_slave_cmd_os;
258  fsoe_slave->fsoe_master_offset = &fsoe_data->fsoe_master_cmd_os;
259  fsoeConf = fsoe_slave->fsoeConf;
260 
261  // alloc crc hal memory
262  if ((fsoe_data->fsoe_crc = hal_malloc(fsoeConf->data_channels * sizeof(lcec_el1918_logic_fsoe_crc_t))) == NULL) {
263  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "hal_malloc() for fsoe_slave %s.%s crc data failed\n", master->name, fsoe_slave->name);
264  return -EIO;
265  }
266  memset(fsoe_data->fsoe_crc, 0, sizeof(lcec_el1918_logic_fsoe_crc_t));
267 
268  // initialize POD entries
269  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x7080 + (fsoe_idx << 4), 0x01, &fsoe_data->fsoe_slave_cmd_os, NULL);
270  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x7080 + (fsoe_idx << 4), 0x02, &fsoe_data->fsoe_slave_connid_os, NULL);
271  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6080 + (fsoe_idx << 4), 0x01, &fsoe_data->fsoe_master_cmd_os, NULL);
272  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6080 + (fsoe_idx << 4), 0x02, &fsoe_data->fsoe_master_connid_os, NULL);
273 
274  // export pins
275  if ((err = lcec_pin_newf_list(fsoe_data, fsoe_pins, LCEC_MODULE_NAME, master->name, slave->name, fsoe_idx)) != 0) {
276  return err;
277  }
278 
279  // map CRC PDOS
280  for (index = 0, crc = fsoe_data->fsoe_crc; index < fsoeConf->data_channels; index++, crc++) {
281  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x7080 + (fsoe_idx << 4), 0x03 + index, &crc->fsoe_slave_crc_os, NULL);
282  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6080 + (fsoe_idx << 4), 0x03 + index, &crc->fsoe_master_crc_os, NULL);
283  if ((err = lcec_pin_newf_list(crc, fsoe_crc_pins, LCEC_MODULE_NAME, master->name, slave->name, fsoe_idx, index)) != 0) {
284  return err;
285  }
286  }
287 
288  fsoe_idx++;
289  fsoe_data++;
290  }
291  }
292 
293  return 0;
294 }

296 void lcec_el1918_logic_read(struct lcec_slave *slave, long period) {
297  lcec_master_t *master = slave->master;
298  lcec_el1918_logic_data_t *hal_data = (lcec_el1918_logic_data_t *) slave->hal_data;
299  uint8_t *pd = master->process_data;
300  lcec_el1918_logic_fsoe_t *fsoe_data;
301  int i, crc_idx;
302  uint8_t std_out;
303  lcec_el1918_logic_fsoe_crc_t *crc;
304  struct lcec_slave *fsoe_slave;
305  const LCEC_CONF_FSOE_T *fsoeConf;
306 
307  *(hal_data->state) = EC_READ_U8(&pd[hal_data->state_os]);
308  *(hal_data->cycle_counter) = EC_READ_U8(&pd[hal_data->cycle_counter_os]);
309 
310  if (hal_data->std_out_count > 0) {
311  std_out = EC_READ_U8(&pd[hal_data->std_out_os]);
312  for (i = 0; i < hal_data->std_out_count; i++) {
313  *(hal_data->std_out_pins[i]) = !!(std_out & (1 << i));
314  }
315  }
316 
317  for (i = 0, fsoe_data = hal_data->fsoe; i < hal_data->fsoe_count; i++, fsoe_data++) {
318  fsoe_slave = fsoe_data->fsoe_slave;
319  fsoeConf = fsoe_slave->fsoeConf;
320  *(fsoe_data->fsoe_master_cmd) = EC_READ_U8(&pd[fsoe_data->fsoe_master_cmd_os]);
321  *(fsoe_data->fsoe_master_connid) = EC_READ_U16(&pd[fsoe_data->fsoe_master_connid_os]);
322  *(fsoe_data->fsoe_slave_cmd) = EC_READ_U8(&pd[fsoe_data->fsoe_slave_cmd_os]);
323  *(fsoe_data->fsoe_slave_connid) = EC_READ_U16(&pd[fsoe_data->fsoe_slave_connid_os]);
324  for (crc_idx = 0, crc = fsoe_data->fsoe_crc; crc_idx < fsoeConf->data_channels; crc_idx++, crc++) {
325  *(crc->fsoe_master_crc) = EC_READ_U16(&pd[crc->fsoe_master_crc_os]);
326  *(crc->fsoe_slave_crc) = EC_READ_U16(&pd[crc->fsoe_slave_crc_os]);
327  }
328  }
329 }

331 void lcec_el1918_logic_write(struct lcec_slave *slave, long period) {
332  lcec_master_t *master = slave->master;
333  lcec_el1918_logic_data_t *hal_data = (lcec_el1918_logic_data_t *) slave->hal_data;
334  uint8_t *pd = master->process_data;
335  uint8_t std_in;
336  int i;
337 
338  if (hal_data->std_in_count > 0) {
339  std_in = 0;
340  for (i = 0; i < hal_data->std_in_count; i++) {
341  if (*(hal_data->std_in_pins[i])) std_in |= (1 << i);
342  }
343  EC_WRITE_U8(&pd[hal_data->std_in_os], std_in);
344  }
345 }
